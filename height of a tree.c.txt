#include <stdio.h>
#include <stdlib.h>

// A structure to represent a binary tree node
struct Node {
    int data;
    struct Node *left;
    struct Node *right;
};

// Function to create a new node
struct Node* newNode(int data) {
    struct Node* node = (struct Node*)malloc(sizeof(struct Node));
    node->data = data;
    node->left = NULL;
    node->right = NULL;
    return node;
}

// Function to calculate the height of a tree using recursion
int height(struct Node* node) {
    // Base case: an empty tree has a height of -1 
    if (node == NULL) {
        return -1;
    } else {
        // Compute the height of each subtree
        int leftHeight = height(node->left);
        int rightHeight = height(node->right);

        // Return the greater of the two heights plus the current node's edge
        if (leftHeight > rightHeight) {
            return(leftHeight + 1);
        } else {
            return(rightHeight + 1);
        }
    }
}

// Main function to demonstrate the height calculation
int main() {
    // Create a sample binary tree
    struct Node* root = newNode(1);
    root->left = newNode(2);
    root->right = newNode(3);
    root->left->left = newNode(4);
    root->left->right = newNode(5);

    // The height of this tree is 2 (path: 1-2-4 or 1-2-5 or 1-3)
    printf("Height of the tree is %d\n", height(root));
    
    // Create another sample tree (skewed)
    struct Node* root2 = newNode(1);
    root2->right = newNode(2);
    root2->right->right = newNode(3);
    root2->right->right->right = newNode(4);

    // The height of this tree is 3 (path: 1-2-3-4)
    printf("Height of the second tree is %d\n", height(root2));

    return 0;
}
